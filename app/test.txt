'use client';

import Image from 'next/image';
import { useState, useEffect } from 'react';

export default function Home() {
    const [articles, setArticles] = useState<any[]>([]);
    const [email, setEmail] = useState('');
    const [password, setPassword] = useState('');
    const [message, setMessage] = useState('');
    const [isLoggedIn, setIsLoggedIn] = useState(false);

    // Fetch raw articles
    useEffect(() => {
        const fetchData = async () => {
            const response = await fetch('https://full-stack-backend-flask.vercel.app/raw');
            const data = await response.json();
            setArticles(data);
        };
        fetchData();
    }, []);

    // Handle Signup
    const handleSignup = async (e: React.FormEvent) => {
        e.preventDefault();
        try {
            const response = await fetch('https://full-stack-backend-flask.vercel.app/signup', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ email, password }),
                credentials: 'include',
            });
            const result = await response.json();
            if (response.ok) {
                setMessage(result.message || 'Signup successful');
            } else {
                setMessage(result.error || 'Signup failed');
            }
        } catch (error) {
            console.error('Signup fetch error:', error); // Log the error for debugging
            setMessage('Failed to connect to the server. Check console for details.');
        }
    };

    // Handle Login
    const handleLogin = async (e: React.FormEvent) => {
        e.preventDefault();
        const response = await fetch('https://full-stack-backend-flask.vercel.app/login', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ email, password }),
            credentials: 'include',
        });
        const result = await response.json();
        if (response.ok) {
            setIsLoggedIn(true);
            setMessage(result.message || 'Login successful');
        } else {
            setMessage(result.error || 'Login failed');
        }
    };

    // Handle Logout
    const handleLogout = async () => {
        const response = await fetch('https://full-stack-backend-flask.vercel.app/logout', {
            method: 'GET',
            credentials: 'include',
        });
        const result = await response.json();
        if (response.ok) {
            setIsLoggedIn(false);
            setMessage(result.message || 'Logout successful');
        } else {
            setMessage('Logout failed');
        }
    };

    return (
        <div className="p-4">
            <h1 className="text-2xl font-bold mb-4">Articles</h1>
            {articles.map((article, index) => (
                <div key={index} className="mb-6 p-4 border rounded-lg shadow">
                    <h2 className="text-xl font-semibold">{article.title}</h2>
                    <p className="text-gray-700 mb-2">{article.abstract}</p>
                    {article.multimedia && article.multimedia.length > 0 && (
                        <Image
                            src={article.multimedia[0].url}
                            alt={article.multimedia[0].caption || 'Article image'}
                            width={article.multimedia[0].width * 3}
                            height={article.multimedia[0].height}
                            className="rounded"
                        />
                    )}
                    <a
                        href={article.url}
                        target="_blank"
                        rel="noopener noreferrer"
                        className="text-blue-500 underline mt-2 block"
                    >
                        Read more
                    </a>
                </div>
            ))}

            {/* Signup Form */}
            <div className="mt-6">
                <h2 className="text-xl font-semibold">Signup</h2>
                <form onSubmit={handleSignup} className="mt-2">
                    <input
                        type="email"
                        value={email}
                        onChange={(e) => setEmail(e.target.value)}
                        placeholder="Email"
                        className="p-2 mb-2 border rounded w-full"
                    />
                    <input
                        type="password"
                        value={password}
                        onChange={(e) => setPassword(e.target.value)}
                        placeholder="Password"
                        className="p-2 mb-2 border rounded w-full"
                    />
                    <button type="submit" className="p-2 bg-blue-500 text-white rounded">
                        Signup
                    </button>
                </form>
                <p>{message}</p>
            </div>

            {/* Login Form */}
            <div className="mt-6">
                <h2 className="text-xl font-semibold">Login</h2>
                <form onSubmit={handleLogin} className="mt-2">
                    <input
                        type="email"
                        value={email}
                        onChange={(e) => setEmail(e.target.value)}
                        placeholder="Email"
                        className="p-2 mb-2 border rounded w-full"
                    />
                    <input
                        type="password"
                        value={password}
                        onChange={(e) => setPassword(e.target.value)}
                        placeholder="Password"
                        className="p-2 mb-2 border rounded w-full"
                    />
                    <button type="submit" className="p-2 bg-blue-500 text-white rounded">
                        Login
                    </button>
                </form>
                <p>{message}</p>
            </div>

            {/* Logout Button */}
            {isLoggedIn && (
                <div className="mt-6">
                    <button onClick={handleLogout} className="p-2 bg-red-500 text-white rounded">
                        Logout
                    </button>
                    <p>{message}</p>
                </div>
            )}
        </div>
    );
}
